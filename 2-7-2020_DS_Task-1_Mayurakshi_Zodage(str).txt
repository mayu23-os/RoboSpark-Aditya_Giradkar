#include<stdio.h>
 struct student
{
  int id,year;
  char name[30];
  float cgpa;
};

int main()
{

  int i=0,j=0;
  struct student st[4];
 while(i!=4)
  {  printf("================================\n");
     printf("Enter details of Student %d\n",i+1);
     printf("Enter the Student ID   ");
     scanf("%d",&st[i].id);
     printf("Enter the name of student   ");
     scanf("%s",st[i].name);
     printf("Enter the year (FY-1 || SY-2 || TY-3 || B.tech-4)  ");
     scanf("%d",&st[i].year);
     printf("Enter CGPA   ");
     scanf("%f",&st[i].cgpa);
     i++;
     fflush(stdin);
  }

   printf("======================================\n");
   printf("ID   ||   Name  ||   Year  ||   CGPA\n");
   printf("======================================\n");
   while(j!=4)
   {
   printf("%d    ||   %s  ||   %d     ||   %.2f\n",st[j].id,st[j].name,st[j].year,st[j].cgpa);
    j++;
  }
}
